rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper: check admin
    function isAdmin() {
      return request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }

    // Users
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update, delete: if request.auth != null && (
        request.auth.uid == userId || isAdmin()
      );
    }

    // Posts / Feed
    match /posts/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid || isAdmin()
      );
    }

    // Stories
    match /stories/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid || isAdmin()
      );
    }

    // Reactions
    match /reactions/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Comments
    match /comments/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid || isAdmin()
      );
    }

    // Messages (story replies)
    match /messages/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Gold transactions
    match /gold_transactions/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Diamond transactions
    match /diamond_transactions/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Admin logs
    match /admin_logs/{id} {
      allow read, write: if request.auth != null && isAdmin();
    }

    // System settings
    match /settings/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && isAdmin();
    }

    // Friendships
    match /friendships/{id} {
      allow read, write: if request.auth != null;
    }
    match /friendships/{id}/chat_messages/{msgId} {
      allow read, write: if request.auth != null;
    }

    // Group rooms
    match /group_rooms/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /group_rooms/{id}/room_messages/{msgId} {
      allow read, write: if request.auth != null;
    }

    // Friend requests
    match /friendRequests/{id} {
      allow read, write: if request.auth != null;
    }

    // Quiz
    match /quiz_questions/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    match /quiz_results/{id} {
      allow read, write: if request.auth != null;
    }

    // Reports
    match /reports/{id} {
      allow read, write: if request.auth != null;
    }

    // Announcements
    match /announcements/{id} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Dice games
    match /dice_games/{id} {
      allow read, write: if request.auth != null;
    }

    // Calls (WebRTC)
    match /calls/{id} {
      allow read, write: if request.auth != null;
    }
    match /calls/{id}/callerCandidates/{cId} {
      allow read, write: if request.auth != null;
    }
    match /calls/{id}/calleeCandidates/{cId} {
      allow read, write: if request.auth != null;
    }

    // ===================== NEW (feed.html / index.html / inbox.html) =====================

    // Communities
    match /communities/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.resource.data.createdBy == request.auth.uid;
      allow update: if request.auth != null; // membersCount updated by any joining/leaving member
      allow delete: if request.auth != null && (
        resource.data.createdBy == request.auth.uid || isAdmin()
      );
    }

    // Follows (doc id pattern: `${followerId}_${followingId}`)
    match /follows/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.resource.data.followerId == request.auth.uid;
      allow delete: if request.auth != null && resource.data.followerId == request.auth.uid;
    }

    // Notifications (created by the actor, read/cleared only by the recipient)
    match /notifications/{id} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.actorId == request.auth.uid;
      allow update: if request.auth != null && resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null && (
        resource.data.userId == request.auth.uid || isAdmin()
      );
    }

    // Saved posts / bookmarks (doc id pattern: `${uid}_${postId}`, private to owner)
    match /savedPosts/{id} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null && resource.data.userId == request.auth.uid;
    }
  }
}
